# Grammar

NProg     ::= [Statement]
Statement ::= Declare

Expr      ::= Term 
            | BinExpr

Term      ::= id
            | int
            | (Expr)

BinExpr   ::= Expr BinOp Expr

NDeclare  ::= let id {= Expr} ;

MULTIPLICATION = >>[-]>[-]<<<[->>+<<]>[->[-<<+>>>+<]>[-<+>]<<]<
