# Grammar

NProg     ::= [Statement]
Statement ::= NDeclare
            | NAssign
            | NRead
            | NPrint

Expr      ::= Term 
            | BinExpr

Term      ::= id
            | int
            | (Expr)

BinExpr   ::= Expr BinOp Expr

NDeclare  ::= let id {= Expr} ;
NAssign   ::= id = Expr ;
NPrint    ::= print Expr ;
NRead     ::= read id ;
